Usage

import { createTabs } from '@coral-css/core/components'

// Create tabs
const tabs = createTabs('#my-tabs', {
  defaultTab: 0,
  activation: 'automatic' // or 'manual'
})

// Select tab programmatically
tabs.selectTab(1)

// Listen for events
tabs.element.addEventListener('coral:tabs:change', (e) => {
  console.log('Tab changed to:', e.detail.index)
})

// Keyboard navigation:
// - ArrowLeft/ArrowRight: Navigate tabs
// - Home: First tab
// - End: Last tab
