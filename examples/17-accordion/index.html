Usage

import { createAccordion } from '@coral-css/core/components'

// Create accordion
const accordion = createAccordion('#my-accordion', {
  allowMultiple: false // Only one panel open at a time
})

// Toggle panel programmatically
accordion.toggle(0)

// Open/close specific panel
accordion.open(1)
accordion.close(1)

// Listen for events
accordion.element.addEventListener('coral:accordion:toggle', (e) => {
  console.log('Panel toggled:', e.detail.index, e.detail.isOpen)
})
